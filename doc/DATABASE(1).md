# DATABASE.md — Schema Reference

PostgreSQL 15 via AWS RDS. ORM: Prisma. Row Level Security (RLS) enforced on all tables.

---

## Conventions

- All IDs are `UUID` generated by `gen_random_uuid()`
- Monetary values stored as `INTEGER` in **centavos** (BRL) — never `FLOAT` or `DECIMAL`
- All timestamps stored in UTC as `TIMESTAMPTZ`
- `created_at` and `updated_at` on every table, auto-managed by Prisma middleware
- `tenant_id` on every user-facing table for SaaS isolation
- Soft deletes via `deleted_at TIMESTAMPTZ NULL` on key tables

## Release 1 schema

### users

```sql
CREATE TABLE users (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  tenant_id     UUID NOT NULL,
  auth0_id      TEXT UNIQUE NOT NULL,
  email         TEXT UNIQUE NOT NULL,
  name          TEXT,
  timezone      TEXT NOT NULL DEFAULT 'America/Sao_Paulo',
  created_at    TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at    TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  deleted_at    TIMESTAMPTZ
);

ALTER TABLE users ENABLE ROW LEVEL SECURITY;
CREATE POLICY users_isolation ON users
  USING (auth0_id = current_setting('app.current_user_id'));
```

### accounts

Manual accounts (banks, wallets, investments) — R1 only has manual; R2 links these to Open Finance.

```sql
CREATE TABLE accounts (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id       UUID NOT NULL REFERENCES users(id),
  name          TEXT NOT NULL,
  type          TEXT NOT NULL CHECK (type IN ('checking','savings','credit','investment','wallet')),
  currency      TEXT NOT NULL DEFAULT 'BRL',
  balance_cents INTEGER NOT NULL DEFAULT 0,
  color         TEXT,
  icon          TEXT,
  is_archived   BOOLEAN NOT NULL DEFAULT FALSE,
  created_at    TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at    TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

ALTER TABLE accounts ENABLE ROW LEVEL SECURITY;
CREATE POLICY accounts_isolation ON accounts
  USING (user_id = (SELECT id FROM users WHERE auth0_id = current_setting('app.current_user_id')));
```

### transactions

```sql
CREATE TABLE transactions (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id         UUID NOT NULL REFERENCES users(id),
  account_id      UUID NOT NULL REFERENCES accounts(id),
  category_id     UUID REFERENCES categories(id),
  amount_cents    INTEGER NOT NULL,           -- negative = expense, positive = income
  description     TEXT NOT NULL,
  notes           TEXT,
  transaction_date DATE NOT NULL,
  type            TEXT NOT NULL CHECK (type IN ('expense','income','transfer')),
  source          TEXT NOT NULL DEFAULT 'manual'
                  CHECK (source IN ('manual','csv_import','bradesco','open_finance')),
  external_id     TEXT,                       -- bank's transaction ID (for dedup)
  is_recurring    BOOLEAN NOT NULL DEFAULT FALSE,
  ai_category_id  UUID REFERENCES categories(id),  -- AI suggestion (may differ from category_id)
  ai_confidence   FLOAT,
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  deleted_at      TIMESTAMPTZ,
  UNIQUE(account_id, external_id)             -- prevent duplicate imports
);

CREATE INDEX idx_transactions_user_date ON transactions(user_id, transaction_date DESC);
CREATE INDEX idx_transactions_category ON transactions(category_id);
CREATE INDEX idx_transactions_source ON transactions(source);

ALTER TABLE transactions ENABLE ROW LEVEL SECURITY;
CREATE POLICY transactions_isolation ON transactions
  USING (user_id = (SELECT id FROM users WHERE auth0_id = current_setting('app.current_user_id')));
```

### budgets

```sql
CREATE TABLE budgets (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id         UUID NOT NULL REFERENCES users(id),
  category_id     UUID NOT NULL REFERENCES categories(id),
  name            TEXT NOT NULL,
  amount_cents    INTEGER NOT NULL,
  period          TEXT NOT NULL CHECK (period IN ('monthly','weekly','yearly')),
  alert_at_pct    INTEGER NOT NULL DEFAULT 80,  -- send alert when X% spent
  is_active       BOOLEAN NOT NULL DEFAULT TRUE,
  created_at      TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at      TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

ALTER TABLE budgets ENABLE ROW LEVEL SECURITY;
CREATE POLICY budgets_isolation ON budgets
  USING (user_id = (SELECT id FROM users WHERE auth0_id = current_setting('app.current_user_id')));
```

### budget_periods

Computed utilisation snapshots (updated by the worker every hour).

```sql
CREATE TABLE budget_periods (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  budget_id       UUID NOT NULL REFERENCES budgets(id),
  period_start    DATE NOT NULL,
  period_end      DATE NOT NULL,
  spent_cents     INTEGER NOT NULL DEFAULT 0,
  alert_sent_80   BOOLEAN NOT NULL DEFAULT FALSE,
  alert_sent_100  BOOLEAN NOT NULL DEFAULT FALSE,
  computed_at     TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  UNIQUE(budget_id, period_start)
);
```

## Release 2 additions

### consents

```sql
CREATE TABLE consents (
  id                UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id           UUID NOT NULL REFERENCES users(id),
  institution       TEXT NOT NULL,             -- e.g. 'bradesco', 'itau'
  institution_id    TEXT NOT NULL,             -- Open Finance participant ID
  consent_id        TEXT UNIQUE NOT NULL,      -- BACEN consent ID
  scopes            TEXT[] NOT NULL,
  status            TEXT NOT NULL DEFAULT 'active'
                    CHECK (status IN ('active','expired','revoked')),
  expires_at        TIMESTAMPTZ NOT NULL,
  revoked_at        TIMESTAMPTZ,
  created_at        TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

ALTER TABLE consents ENABLE ROW LEVEL SECURITY;
CREATE POLICY consents_isolation ON consents
  USING (user_id = (SELECT id FROM users WHERE auth0_id = current_setting('app.current_user_id')));
```

### open_finance_tokens

Access and refresh tokens — stored encrypted via pgcrypto.

```sql
CREATE TABLE open_finance_tokens (
  id                UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  consent_id        UUID NOT NULL REFERENCES consents(id),
  access_token_enc  TEXT NOT NULL,    -- encrypted
  refresh_token_enc TEXT NOT NULL,    -- encrypted
  expires_at        TIMESTAMPTZ NOT NULL,
  updated_at        TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
```

### bank_accounts (R2 — linked to Open Finance)

Adds `consent_id` foreign key to the existing `accounts` table via migration:

```sql
ALTER TABLE accounts ADD COLUMN consent_id UUID REFERENCES consents(id);
ALTER TABLE accounts ADD COLUMN external_account_id TEXT;
ALTER TABLE accounts ADD COLUMN last_synced_at TIMESTAMPTZ;
```

### market_quotes

```sql
CREATE TABLE market_quotes (
  ticker          TEXT NOT NULL,
  quote_date      DATE NOT NULL,
  open_cents      INTEGER NOT NULL,
  high_cents      INTEGER NOT NULL,
  low_cents       INTEGER NOT NULL,
  close_cents     INTEGER NOT NULL,
  volume          BIGINT,
  PRIMARY KEY (ticker, quote_date)
) PARTITION BY RANGE (quote_date);

-- Create partitions by year
CREATE TABLE market_quotes_2024 PARTITION OF market_quotes
  FOR VALUES FROM ('2024-01-01') TO ('2025-01-01');
CREATE TABLE market_quotes_2025 PARTITION OF market_quotes
  FOR VALUES FROM ('2025-01-01') TO ('2026-01-01');
```

## Prisma schema snippet

```prisma
// packages/api/prisma/schema.prisma

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

model User {
  id           String        @id @default(dbgenerated("gen_random_uuid()")) @db.Uuid
  tenantId     String        @db.Uuid
  auth0Id      String        @unique
  email        String        @unique
  name         String?
  timezone     String        @default("America/Sao_Paulo")
  createdAt    DateTime      @default(now())
  updatedAt    DateTime      @updatedAt
  deletedAt    DateTime?
  accounts     Account[]
  transactions Transaction[]
  budgets      Budget[]
  consents     Consent[]
}

model Transaction {
  id              String    @id @default(dbgenerated("gen_random_uuid()")) @db.Uuid
  userId          String    @db.Uuid
  accountId       String    @db.Uuid
  categoryId      String?   @db.Uuid
  amountCents     Int
  description     String
  transactionDate DateTime  @db.Date
  type            String
  source          String    @default("manual")
  externalId      String?
  aiCategoryId    String?   @db.Uuid
  aiConfidence    Float?
  createdAt       DateTime  @default(now())
  updatedAt       DateTime  @updatedAt
  deletedAt       DateTime?
  user            User      @relation(fields: [userId], references: [id])
  account         Account   @relation(fields: [accountId], references: [id])
  category        Category? @relation(fields: [categoryId], references: [id])
}
```

```bash
# Create a new migration
cd packages/api

# Apply pending migrations in production
pnpm prisma migrate deploy

# View migration history
pnpm prisma migrate status
```
